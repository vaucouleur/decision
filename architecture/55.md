## Add this macro to `crates/smt-tests/src/eqshare_macros.rs`

```rust
#[macro_export]
macro_rules! assert_eqshare_events_empty {
    ($events:expr $(,)?) => {{
        let events = $events;
        if !events.is_empty() {
            panic!(
                "expected no eqshare events, but got {}.\nfirst events:\n{}",
                events.len(),
                $crate::eqshare_macros::dump(events)
            );
        }
    }};
}
```

## Example usage

```rust
use crate::{assert_eqshare_events_empty};

let mut sess = common::make_session(SharingConfig { uf_to_dl: false, dl_to_uf: false });
let _inst = setup_ping_pong_instance(&mut sess)?;
assert_eq!(sess.check_sat(), CheckSat::Sat);

let events = sess.take_eqshare_events();
assert_eqshare_events_empty!(&events);
```
